    class Fibs:
        def __init__(self):
                self.a = 0
                self.b = 0
        def __next__(self):
                self.a,self.b = self.b,self.a+self.b
                return self.a
        def __iter__(self):
                return self

fib = Fibs()

for f in fib:
    if f > 1000:
        print(f)
